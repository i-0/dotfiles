* Package management
#+BEGIN_SRC emacs-lisp
    (require 'package)

    (add-to-list 'package-archives '("marmalade" . "http://marmalade-repo.org/packages/") t)
    (add-to-list 'package-archives '("tromey"    . "http://tromey.com/elpa/") t)
    (add-to-list 'package-archives '("melpa"     . "http://melpa.milkbox.net/packages/") t)
    (add-to-list 'package-archives '("org"       . "http://orgmode.org/elpa/") t)
    (add-to-list 'package-archives '("gnu"       . "http://elpa.gnu.org/packages/") t)

    (package-initialize)

    (when (not package-archive-contents)
      (package-refresh-contents))

    (defvar my-packages '(
                          ;; evil
                          evil
                          evil-leader
                          evil-magit
                          evil-nerd-commenter
                          evil-org
                          evil-surround
                          evil-visualstar

                          ;; project management
                          projectile

                          ;; git integration
                          magit

                          ;; helm
                          helm
                          helm-projectile

                          ;; misc tools
                          restclient
                          urlenc
                          expand-region
                          ;smart-tabs-mode
                          ;exec-path-from-shell

                          ;; plantuml integration
                          plantuml-mode

                          ;; java, eclipse integration
                          eclim

                          ;; javascript
                          js2-mode
                          ac-js2

                          ;; clojure (experimental)
                          cider
                          clojure-mode
                          paredit
                          clojure-mode-extra-font-locking
                          rainbow-delimiters

                          ;; color-theme
                          color-theme-sanityinc-solarized
                          ))

    (dolist (p my-packages)
      (when (not (package-installed-p p))
        (package-install p)))
#+END_SRC

* Setup expand-region
From here: http://emacsrocks.com/e09.html

Select expanding region in visual mode.
- selection within braces or quotes
- word, big word, line , paragraph, ... up to whole buffer
| command           | function         | effect                                  |
| /                 | <>               |                                         |
|-------------------+------------------+-----------------------------------------|
| , v (normal mode) | er/expand-region | expand from word, big word, line to ... |

#+BEGIN_SRC emacs-lisp
  (require 'expand-region)
  (global-set-key (kbd "C-@") 'er/expand-region)
  (evil-leader/set-key "v" 'er/expand-region)
#+END_SRC
